class Stack {
  length = 0;

  constructor() {
    this.stack = [];
  }
  /**
   * empty method checks to see if the Stack is empty.
   * @method
   * @returns {boolean} true, if empty. False otherwise
   */
  empty() {
    return this.length == 0;
  }
  /**
   * peek method allows you to take look at the last item pushed to the stack
   * @method
   * @returns
   */
  peek() {
    return this.stack[this.stack.length - 1];
  }
  /**
   * pop method removed the last item pushed into the stack.
   * @method
   * @returns
   */
  pop() {
    if (this.length > 0) {
      this.length--;
      return this.stack.pop();
    }
  }
  /**
   * push method adds a new item to the stack
   * @method
   * @param {Any} item
   */
  push(item) {
    this.length++;
    this.stack.push(item);
  }

  toString() {
    return this.stack.toString();
  }
}

export { Stack };
